    function playSoundFromDataUrl(dataUrl) {
      try {
        var ctx = _ensureAudioCtx();

        var base64 = dataUrl.split(',')[1];
        var binary = atob(base64);
        var bytes = new Uint8Array(binary.length);
        for (var i = 0; i < binary.length; i++) bytes[i] = binary.charCodeAt(i);

          var source = ctx.createBufferSource();
          source.buffer = audioBuffer;
          source.connect(ctx.destination);
          source.start(0);

    // Phase 7 (M-4): postMessage-based security event reporting — details never in URL
    function _reportSecurityEvent(eventType, details) {
      // Surrendered-tab guard: don't fire GAS requests when this tab has yielded
      // to another tab. The surrendered tab should be idle — no network activity.
      if (_tabSurrendered) return;
      // Pre-auth guard: don't fire GAS requests when no session exists.

      if (!baseUrl) return;

      var frame = document.createElement('iframe');
      frame.style.cssText = 'display:none;width:0;height:0;border:0';
      frame.src = baseUrl + '?action=securityEvent';
      document.body.appendChild(frame);
      var _seHandled = false;
      var _seEventType = eventType;
      var _seDetails = details;
      var _seHandler = function(event) {
        if (_seHandled) return;
        if (!event.data || event.data.type !== 'gas-security-event-ready') return;
        _seHandled = true;
        window.removeEventListener('message', _seHandler);
        event.source.postMessage({
          type: 'security-event-report',
          eventType: _seEventType,
          details: _seDetails,
          timestamp: Date.now()
        }, event.origin);
      };
      window.addEventListener('message', _seHandler);
      // Cleanup after 10 seconds
      setTimeout(function() {
        if (!_seHandled) {
          _seHandled = true;
          window.removeEventListener('message', _seHandler);
        }
        if (frame.parentNode) frame.parentNode.removeChild(frame);

    // ── HMAC Key Import (Web Crypto API) ──
    // Called once when gas-session-created delivers the messageKey.
    // Key is imported as non-extractable — even XSS cannot read it back.
    // Permissions: verify-only (cannot sign forged messages).
    // Phase 5 (H-3): _hmacKeySet flag provides defense-in-depth — once set,
    // external code cannot overwrite the key via a forged gas-session-created.
    var _hmacKeySet = false;
    async function _importHmacKey(rawKey) {
      if (_hmacKeySet) return _hmacKey;  // Already set — reject overwrites
      var encoder = new TextEncoder();
      var keyData = encoder.encode(rawKey);
      var key = await crypto.subtle.importKey(
        'raw',
        keyData,
        { name: 'HMAC', hash: 'SHA-256' },
        false,   // non-extractable
        ['verify'] // verify-only — cannot sign
      );
      _hmacKey = key;
      _hmacKeySet = true;
      return key;
    }

    // ── HMAC-SHA256 Verification (Web Crypto API) ──
    // Verifies a hex-encoded HMAC-SHA256 signature against the payload.
    async function _verifyHmacSha256(data, hexSig) {
      if (!_hmacKey) return false;
      // Reconstruct payload identically to how GAS computed it (sorted keys, exclude _sig)
      var copy = {};
      var keys = Object.keys(data).sort();
      for (var i = 0; i < keys.length; i++) {
        if (keys[i] !== '_sig') copy[keys[i]] = data[keys[i]];
      }
      var payload = JSON.stringify(copy);
      // Decode hex signature to ArrayBuffer
      var sigBytes = new Uint8Array(hexSig.match(/.{1,2}/g).map(function(byte) {
        return parseInt(byte, 16);
      }));
      var encoder = new TextEncoder();
      var payloadBytes = encoder.encode(payload);
      return crypto.subtle.verify('HMAC', _hmacKey, sigBytes, payloadBytes);
    }

    // ── HMAC-SHA256 Signature Verification ──
    // DJB2 legacy removed — all messages now signed server-side with HMAC-SHA256.
    async function _verifyMessageSignature(data) {
      if (!_hmacKey) return true;  // No key yet — allowlist is defense
      var sig = data._sig;
      if (!sig) return false;  // Key exists but no signature — reject
      return _verifyHmacSha256(data, sig);
    }

          // before the GAS backend has validated the token.
          _pendingSessionShow = { email: data.email, displayName: data.displayName, role: data.role };
          // Normal sign-in flow (OAuth) — reload iframe with session token
          // so the app UI loads (first load was the token exchange page).
          // The ?session path does NOT send gas-session-created — it sends
          // gas-auth-ok (with messageKey) directly. Key import is handled there.
          var gasApp = document.getElementById('gas-app');
          var baseUrl = (gasApp && gasApp.dataset.baseUrl) || '';
          if (gasApp && baseUrl) {
            _expectingSession = true;
            gasApp.src = baseUrl + '?session=' + encodeURIComponent(data.sessionToken);
          }
        } else {
          // Surface specific misconfiguration errors so the admin knows what to fix
          var authError = data.error || '';
          var authMsg = 'Access denied. Contact your administrator.';
          if (authError === 'hmac_secret_missing') {
            authMsg = 'Setup required: HMAC_SECRET is not configured in Script Properties. '
              + 'Open GAS Editor → Project Settings → Script Properties → Add: HMAC_SECRET = <random-64-char-hex-string> (run: openssl rand -hex 32). Then redeploy.';
          } else if (authError === 'domain_not_configured') {
            authMsg = 'Setup required: Domain restriction is enabled but ALLOWED_DOMAINS is empty. '
              + 'Set ALLOWED_DOMAINS in the GAS preset config or disable ENABLE_DOMAIN_RESTRICTION via PROJECT_OVERRIDES.';
          } else if (authError === 'domain_not_allowed') {
            authMsg = 'Access denied: your email domain is not in the allowed list. Contact your administrator.';
          } else if (authError === 'account_locked') {
            authMsg = 'Account temporarily locked due to too many failed attempts. Contact your administrator.';
          } else if (authError === 'rate_limited') {
            authMsg = 'Too many sign-in attempts. Please wait a few minutes and try again.';
          }
          showAuthWall(authMsg);
        }
      }

      // GAS iframe says auth is needed (session expired/invalid)
      if (data.type === 'gas-needs-auth') {
        if (_expectingSession) {
          // Stale gas-needs-auth — the iframe was still loading a prior URL
          // while we already set a new src with the session token. Ignore
          // this message and wait for the session-URL response instead.
          // NOTE: _expectingSession is NOT set on page-load resume — only on
          // mid-session navigations (gas-session-created, Use Here) where
          // prior iframe content could emit a stale gas-needs-auth.
          _expectingSession = false;
          return;
        }
        _directSessionLoad = false;  // Clear in case "Use Here" token was invalid
        clearSession();
        stopCountdownTimers();
        stopHeartbeat();
        sessionStorage.setItem(SESSION_EXPIRED_FLAG, '1');
        // Differentiate admin sign-out from normal expiration
        var evReason = data.evictionReason || '';
        var authMsg = 'Session expired. Please sign in again.';
        if (evReason === 'admin_signout') {
          authMsg = 'An administrator ended your session. Please sign in again.';
        } else if (evReason === 'new_sign_in') {
          authMsg = 'Your session was ended (signed in elsewhere). Please sign in again.';
        }
        showAuthWall(authMsg);
      }

      // GAS iframe confirms auth is OK (for returning users)
      if (data.type === 'gas-auth-ok') {
        _expectingSession = false;
        // Phase 5: Import messageKey if delivered via gas-auth-ok (the ?session path
        // includes it). This handles "Use Here" reclaim, tab duplicate, and page refresh
        // where gas-session-created is not sent. Only import if no key is set yet.
        if (data.messageKey && !_hmacKeySet) {
          _importHmacKey(data.messageKey).catch(function(err) {
            console.error('HMAC key import from gas-auth-ok failed:', err);
          });
        }
        // RBAC: Update role/permissions from server (covers session resume where
        // gas-session-created is not sent, and role changes between sessions)
        if (data.role) {
          getStorage().setItem(ROLE_KEY, data.role);
          if (data.permissions) {
            getStorage().setItem(PERMISSIONS_KEY, JSON.stringify(data.permissions));
          }
        }
        // Phase 3: IP forwarding removed — uncomment to re-enable
        // if (!_ipForwardedToGas && _clientIp && HTML_CONFIG.ENABLE_IP_LOGGING) {
        //   _ipForwardedToGas = true;
        //   var gasIpFrame = document.getElementById('gas-app');
        //   if (gasIpFrame && gasIpFrame.contentWindow) {
        //     gasIpFrame.contentWindow.postMessage({type: 'host-client-ip', ip: _clientIp}, event.origin);
        //   }
        // }
        // "Use Here" hides the iframe before reload — re-show it now that
        // the GAS app confirmed the session is valid.
        if (_directSessionLoad) {
          _directSessionLoad = false;
          var gasAppReshow = document.getElementById('gas-app');
          if (gasAppReshow) gasAppReshow.style.visibility = '';
          // The "Use Here" handler deferred these to avoid premature UI —
          // now that auth is confirmed, show the app and start timers.
          var session = loadSession();
          showApp(session.email, null, session.role);
          startCountdownTimers();
          startHeartbeat();
          // The server may report needsReauth because the server-side
          // session was close to expiring, but the client just reset its
          // rolling timer and the heartbeat will extend the server session
          // momentarily.  Suppress the false warning.
          _expiryWarningShown = false;
          hideAllWarningBanners();
        } else if (_pendingSessionShow) {
          // Deferred UI show — gas-session-created, page-load resume, or cross-tab
          // sync deferred the showApp call until the GAS backend confirmed the
          // session is valid via gas-auth-ok.
          showApp(_pendingSessionShow.email, _pendingSessionShow.displayName, _pendingSessionShow.role);
          startCountdownTimers();
          startHeartbeat();
          broadcastTabClaim();
          _pendingSessionShow = null;
        } else if (data.needsReauth) {
          showSessionWarning('Session expiring soon \u2014 interact with the page to stay signed in.');
        }
      }

      // Heartbeat OK — server extended the session
      if (data.type === 'gas-heartbeat-ok') {
        _heartbeatInFlight = false;
        if (_heartbeatInFlightTimer) { clearTimeout(_heartbeatInFlightTimer); _heartbeatInFlightTimer = null; }
        _heartbeatIdle = true;  // reset to idle after extend — next tick decides if still active
        // Reset the client-side session start time to match the server reset
        getStorage().setItem(SESSION_START_KEY, String(Date.now()));
        // Hide session warning and reset flag — session was extended
        _expiryWarningShown = false;
        hideSessionWarning();
      }

      // Heartbeat failed — session already expired on server
      if (data.type === 'gas-heartbeat-expired') {
        _heartbeatInFlight = false;
        if (_heartbeatInFlightTimer) { clearTimeout(_heartbeatInFlightTimer); _heartbeatInFlightTimer = null; }
        stopHeartbeat();
        var hbEl2 = document.getElementById('timer-heartbeat');
        if (hbEl2) {
          hbEl2.textContent = 'expired';
          hbEl2.className = 'timer-value critical';
        }
        stopCountdownTimers();

        // Admin sign-out: an administrator ended this session
        if (data.reason === 'admin_signout') {
          clearSession();
          sessionStorage.setItem(SESSION_EXPIRED_FLAG, '1');
          showAuthWall('An administrator ended your session. Please sign in again.');
          return;
        }

        // Cross-device eviction: another device signed in
        if (data.reason === 'new_sign_in' && HTML_CONFIG.CROSS_DEVICE_ENFORCEMENT) {
          _evictedByRemote = true;
          _tabSurrendered = true;
          clearSession();
          hideAllWarningBanners();

          var takeoverWall = document.getElementById('tab-takeover-wall');
          if (takeoverWall) {
            var takeoverText = takeoverWall.querySelector('p');
            if (takeoverText) {
              takeoverText.textContent =
                'This session was started on another device or browser. '
                + 'Only one session can be active at a time.';
            }
            var useHereBtn = document.getElementById('tab-use-here-btn');
            if (useHereBtn) {
              useHereBtn.textContent = 'Sign In Here';
            }
            takeoverWall.style.display = 'flex';
          }
          return;  // Don't show auth wall — show takeover overlay instead
        }

        // Normal expiration (timeout, absolute timeout, corrupt session, etc.)
        clearSession();
        sessionStorage.setItem(SESSION_EXPIRED_FLAG, '1');
        showAuthWall('Session expired. Please sign in again.');
      }

      // GAS data operation detected invalid session (Phase 3 — server-side validation gate)
      if (data.type === 'gas-session-invalid') {
        stopHeartbeat();
        stopCountdownTimers();
        clearSession();
        sessionStorage.setItem(SESSION_EXPIRED_FLAG, '1');
        var reason = data.reason || 'SESSION_EXPIRED';
        var msg = 'Session expired during data operation. Please sign in again.';
        if (reason === 'SESSION_EVICTED') {
          msg = 'Your session was ended (signed in elsewhere). Please sign in again.';
        } else if (reason === 'SESSION_INTEGRITY_VIOLATION') {
          msg = 'Session integrity check failed. Please sign in again.';
        }
        showAuthWall(msg);
      }

      // GAS iframe user activity — trigger immediate heartbeat to catch expired sessions
      // before the user submits data (prevents data loss on evicted sessions)
      if (data.type === 'gas-user-activity') {
        // Bridge the cross-origin gap: host page activity listeners can't see
        // events inside the GAS iframe, so update the heartbeat activity
        // timestamp here.  The regular interval tick and the urgent <30s
        // heartbeat will use this to decide when to send.
        _lastHeartbeatActivity = Date.now();
        _heartbeatIdle = false;
      }

      // GAS sign-out confirmation
      if (data.type === 'gas-signed-out') {
        clearSession();
        stopCountdownTimers();
        stopHeartbeat();
        showAuthWall('You have been signed out.');
      }
    }

    // =============================================
    // AUTH — Cross-Tab Session Sync
    // =============================================
    // Two mechanisms handle cross-tab sign-out, matched to their storage type:
    //
    // 1. localStorage (standard preset): the browser's `storage` event fires automatically
    //    in other same-origin tabs when localStorage is modified. Sign-out clears the session
    //    key from localStorage, which triggers the event in other tabs for free — no explicit
    //    signaling code needed. This is the most elegant approach for localStorage because
    //    cross-tab sync is a natural byproduct of the storage mechanism itself.
    //
    // 2. sessionStorage (hipaa preset): sessionStorage is per-tab by design — changes never
    //    fire cross-tab `storage` events. However, duplicating a tab clones its sessionStorage
    //    (per HTML spec), so a duplicated tab inherits the session. Without explicit signaling,
    //    signing out of one tab leaves duplicated tabs still authenticated. BroadcastChannel
    //    solves this: sign-out broadcasts a message, and all same-origin tabs receive it and
    //    clear their own sessions. BroadcastChannel is purpose-built for cross-tab communication
    //    and works regardless of storage type.
    //
    // Each mechanism is the right tool for its storage type — localStorage gets free sync via
    // storage events, sessionStorage needs explicit signaling via BroadcastChannel.

    // --- Mechanism 1: storage event (localStorage / standard preset) ---
    window.addEventListener('storage', function(e) {
      // Login sync: another tab saved a session token
      if (e.key === SESSION_KEY && !e.oldValue && e.newValue) {
        var session = loadSession();
        if (session.token) {
          // Defer showApp until gas-auth-ok confirms the session server-side.
          // The iframe will load with the session token and the GAS backend will
          // validate it — only then does the auth wall come down.
          _pendingSessionShow = { email: session.email, displayName: null, role: session.role };
          getStorage().setItem(SESSION_START_KEY, String(Date.now()));
          getStorage().setItem(ABSOLUTE_START_KEY, String(Date.now()));
          var gasApp = document.getElementById('gas-app');
          var baseUrl = (gasApp && gasApp.dataset.baseUrl) || '';
          if (gasApp && baseUrl) {
            _expectingSession = true;
            gasApp.src = baseUrl + '?session=' + encodeURIComponent(session.token);
          }
        }
      }

      // Logout sync: another tab cleared the session token
      if (e.key === SESSION_KEY && e.oldValue && !e.newValue) {
        clearSession();
        stopCountdownTimers();
        stopHeartbeat();
        showAuthWall('You have been signed out in another tab.');
      }
    });

    // --- Mechanism 2: BroadcastChannel (sessionStorage / hipaa preset + single-tab) ---
    // Active when using sessionStorage (sign-out sync) or when single-tab enforcement is on.
    var _signOutChannel = null;
    if ((HTML_CONFIG.STORAGE_TYPE === 'sessionStorage' || HTML_CONFIG.SINGLE_TAB_ENFORCEMENT)
        && typeof BroadcastChannel !== 'undefined') {
      _signOutChannel = new BroadcastChannel(_pageName + '-auth-sign-out');
      _signOutChannel.onmessage = function(e) {
        if (!e.data) return;
        // Tab roll call: another tab is asking "who's there?" — respond with presence
        if (e.data.type === 'tab-roll-call' && e.data.tabId !== _tabId) {
          _signOutChannel.postMessage({ type: 'tab-present', tabId: _tabId });
        }
        // Tab roll call response: collect presence for count display
        if (e.data.type === 'tab-present' && e.data.tabId && e.data.tabId !== _tabId) {
          _rollCallResponses[e.data.tabId] = true;
          updateTakeoverTabCount();
        }
        if (e.data.type === 'sign-out') {
          // Hide tab-takeover overlay if visible (sign-out takes priority)
          document.getElementById('tab-takeover-wall').style.display = 'none';
          _tabSurrendered = false;
          clearSession();
          stopCountdownTimers();
          stopHeartbeat();
          showAuthWall('You have been signed out in another tab.');
        }
        // Single-tab enforcement: another tab is claiming the session
        if (e.data.type === 'tab-claim' && HTML_CONFIG.SINGLE_TAB_ENFORCEMENT) {
          if (e.data.tabId !== _tabId) {
            _tabSurrendered = true;
            _evictedByRemote = false;  // Ensure correct overlay text for same-browser claim
            // Preserve the absolute start time — stopCountdownTimers() clears
            // it, but "Use Here" needs to continue the original absolute timer
            // rather than restarting it.
            var savedAbsoluteStart = getStorage().getItem(ABSOLUTE_START_KEY);
            stopCountdownTimers();
            if (savedAbsoluteStart) {
              getStorage().setItem(ABSOLUTE_START_KEY, savedAbsoluteStart);
            }
            stopHeartbeat();
            hideAllWarningBanners();
            // Phase 4 (H-1): No credentials in broadcast — each tab already has
            // its own session in sessionStorage. When "Use Here" is clicked, the
            // tab reclaims using its stored token (see Use Here handler below).
            // Reset overlay text to same-browser message (in case it was changed
            // by a previous cross-device eviction).
            var takeoverWall = document.getElementById('tab-takeover-wall');
            if (takeoverWall) {
              var takeoverText = takeoverWall.querySelector('p');
              if (takeoverText) {
                takeoverText.textContent =
                  'This session is active in another tab. Only one tab can be active at a time.';
              }
              var useHereBtn = document.getElementById('tab-use-here-btn');
              if (useHereBtn) {
                useHereBtn.textContent = 'Use Here';
              }
            }
            document.getElementById('tab-takeover-wall').style.display = 'flex';
            // Ask other tabs to identify themselves — responses update the
            // overlay text in real time via updateTakeoverTabCount().
            startTabRollCall();
          }
        }
      };
    }

    // Tab roll call — ask other tabs to identify themselves.
    // Called once when the tab-takeover overlay is shown. Responses update the
    // overlay text in real time via updateTakeoverTabCount().
    function startTabRollCall() {
      _rollCallResponses = {};
      if (_signOutChannel) {
        _signOutChannel.postMessage({ type: 'tab-roll-call', tabId: _tabId });
      }
    }

    // Update the tab-takeover overlay text with the current response count.
    function updateTakeoverTabCount() {
      var takeoverWall = document.getElementById('tab-takeover-wall');
      if (!takeoverWall || takeoverWall.style.display === 'none') return;
      var takeoverText = takeoverWall.querySelector('p');
      if (!takeoverText) return;
      var count = Object.keys(_rollCallResponses).length;
      var tabNote = count > 1
        ? ' (' + count + ' other tabs have this page open)'
        : '';
      if (_evictedByRemote) {
        takeoverText.textContent =
          'Your session was started on another device. Sign in again to use it here.' + tabNote;
      } else {
        takeoverText.textContent =
          'This session is active in another tab. Only one tab can be active at a time.' + tabNote;
      }
    }

    // Broadcast tab-claim to other tabs (single-tab enforcement)
    // Phase 4 (H-1): Only broadcast the event + tabId — never send credentials
    // over BroadcastChannel. Each tab reads its own credentials from sessionStorage.
    function broadcastTabClaim() {
      if (HTML_CONFIG.SINGLE_TAB_ENFORCEMENT && _signOutChannel) {
        _signOutChannel.postMessage({
          type: 'tab-claim',
          tabId: _tabId
        });
      }
    }

    function showApp(email, displayName, role) {

    /**
     * Apply UI element gating based on data-requires-permission and data-requires-role attributes.
     * - data-requires-permission="permName": hidden unless the user's permissions array includes permName.
     * - data-requires-role="roleName": hidden unless the user's role matches roleName.
     * Both attributes can coexist on the same element (both conditions must be met).
     * No spreadsheet tab needed — just add the attribute to any HTML element.
     */
    function applyUIGating() {
      var session = loadSession();
      var perms = session.permissions || [];
      var role = session.role || '';
      var gated = document.querySelectorAll('[data-requires-permission], [data-requires-role]');
      for (var i = 0; i < gated.length; i++) {
        var el = gated[i];
        var visible = true;
        var reqPerm = el.getAttribute('data-requires-permission');
        if (reqPerm && perms.indexOf(reqPerm) === -1) visible = false;
        var reqRole = el.getAttribute('data-requires-role');
        if (reqRole && role !== reqRole) visible = false;
        el.style.display = visible ? 'inline-block' : 'none';
      }
    }

    function toggleTimers(e) {
      if (e) e.stopPropagation();
      var container = document.getElementById('auth-timers');
      var body = document.getElementById('timer-body');
      var icon = document.getElementById('timer-toggle-icon');

    var SERVER_SESSION_DURATION = 3600;    // seconds — must match GAS PRESETS.standard.SESSION_EXPIRATION (1hr)
    var ABSOLUTE_SESSION_DURATION = 28800; // seconds — must match GAS PRESETS.standard.ABSOLUTE_SESSION_TIMEOUT (8hr)

    // ── Phase 6 (H-4): Immutable Session Timers ──
    // Closure-scoped timer enforcer — prevents client-side modification of session
    // duration values via DevTools console. HIPAA §164.312(a)(2)(iii) requires
    // automatic logoff; tampered timers could keep a session visually alive on a
    // shared workstation between heartbeats. The GAS server independently enforces
    // session expiry (true authority), but immutable client timers close the
    // visual-state gap between heartbeats.
    var _sessionTimerEnforcer = (function() {
      var _serverDuration = null;
      var _absoluteDuration = null;
      var _locked = false;
      return {
        lock: function(serverDur, absoluteDur) {
          if (_locked) return;  // Already locked — ignore attempts to change
          _serverDuration = serverDur;
          _absoluteDuration = absoluteDur;
          _locked = true;
        },
        getServerDuration: function() { return _serverDuration; },
        getAbsoluteDuration: function() { return _absoluteDuration; },
        isLocked: function() { return _locked; },
        reset: function() {
          // Only called during clearSession() — resets for the next sign-in
          _serverDuration = null;
          _absoluteDuration = null;
          _locked = false;
        }
      };
    })();

    /**
     * Phase 6 (H-4): Lock session timer constants using Object.defineProperty.
     * Belt-and-suspenders: both Object.defineProperty (prevents window.X = Y)
     * and the closure-scoped enforcer (prevents delete + redefine bypass).
     * Called once from gas-session-created handler after receiving server durations.
     */
    function _lockSessionTimers(serverDuration, absoluteDuration) {
      // Lock the closure-scoped enforcer first (primary defense)
      _sessionTimerEnforcer.lock(serverDuration, absoluteDuration);

      // Also lock global variables via Object.defineProperty (secondary defense)
      // var-declared globals are configurable:false but writable:true — we flip
      // writable to false so console assignments silently fail in non-strict mode.
      // Check writable (not configurable) since var declarations are already
      // non-configurable. Skip if already locked (writable:false).
      var desc = Object.getOwnPropertyDescriptor(window, 'SERVER_SESSION_DURATION');
      if (!desc || desc.writable) {
        Object.defineProperty(window, 'SERVER_SESSION_DURATION', {
          value: serverDuration,
          writable: false
        });
      }
      var descAbs = Object.getOwnPropertyDescriptor(window, 'ABSOLUTE_SESSION_DURATION');
      if (!descAbs || descAbs.writable) {
        Object.defineProperty(window, 'ABSOLUTE_SESSION_DURATION', {
          value: absoluteDuration,
          writable: false
        });
      }
    }
    var SESSION_START_KEY = _pageName + '_gas_session_start';
    var ABSOLUTE_START_KEY = _pageName + '_gas_absolute_start';
    var _timerInterval = null;

    function formatCountdown(seconds) {
      if (seconds <= 0) return '0:00';
      var h = Math.floor(seconds / 3600);
      var m = Math.floor((seconds % 3600) / 60);
      var s = Math.floor(seconds % 60);
      if (h > 0) {
        return h + ':' + (m < 10 ? '0' : '') + m + ':' + (s < 10 ? '0' : '') + s;
      }
      return m + ':' + (s < 10 ? '0' : '') + s;
    }

    function startCountdownTimers() {
      var timersEl = document.getElementById('auth-timers');
      var sessionEl = document.getElementById('timer-session');
      var absoluteEl = document.getElementById('timer-absolute');
      var heartbeatEl = document.getElementById('timer-heartbeat');
      var pillLabel = document.getElementById('timer-pill-label');

      // Phase 6 (H-4): Lock timer values if not already locked.
      // gas-session-created locks them on fresh sign-in (with server-provided values).
      // On session resume (gas-auth-ok), gas-session-created never fires — lock here
      // as a catch-all using the var declarations. No-op if already locked.
      _lockSessionTimers(SERVER_SESSION_DURATION, ABSOLUTE_SESSION_DURATION);
      // Read timer values from the closure-scoped enforcer (tamper-proof).
      var _serverDur = _sessionTimerEnforcer.getServerDuration() || SERVER_SESSION_DURATION;
      var _absDur = _sessionTimerEnforcer.getAbsoluteDuration() || ABSOLUTE_SESSION_DURATION;

      // Store session start time if not already stored
      var storage = getStorage();
      if (!storage.getItem(SESSION_START_KEY)) {
        storage.setItem(SESSION_START_KEY, String(Date.now()));
      }
      // Store absolute start time if not already stored (set once, never reset)
      if (!storage.getItem(ABSOLUTE_START_KEY)) {
        storage.setItem(ABSOLUTE_START_KEY, String(Date.now()));
      }

      // Show absolute timer status (Phase 6: use enforcer value)
      if (_absDur > 0) {
        absoluteEl.textContent = formatCountdown(_absDur);
        absoluteEl.className = 'timer-value';
      } else {
        absoluteEl.textContent = 'off';
        absoluteEl.className = 'timer-value disabled';
      }

      // Show heartbeat status
      if (HTML_CONFIG.ENABLE_HEARTBEAT) {
        heartbeatEl.textContent = 'waiting';
        heartbeatEl.className = 'timer-value disabled';
      } else {
        heartbeatEl.textContent = 'off';
        heartbeatEl.className = 'timer-value disabled';
      }

      timersEl.style.display = 'flex';

      if (_timerInterval) clearInterval(_timerInterval);
      _timerInterval = setInterval(function() {
        // Phase 6 (H-4): Read from enforcer each tick (tamper-proof source)
        var _curServerDur = _sessionTimerEnforcer.getServerDuration() || SERVER_SESSION_DURATION;
        var _curAbsDur = _sessionTimerEnforcer.getAbsoluteDuration() || ABSOLUTE_SESSION_DURATION;
        var startTime = parseInt(storage.getItem(SESSION_START_KEY) || '0', 10);
        var elapsed = (Date.now() - startTime) / 1000;
        var remaining = _curServerDur - elapsed;

        // Urgent heartbeat: if <30s remain and the user has been active recently,
        // send a heartbeat immediately instead of waiting for the next 30s interval.
        // This closes the gap where activity in the last 30s before expiry goes unnoticed.
        if (HTML_CONFIG.ENABLE_HEARTBEAT && _heartbeatInterval && remaining > 0 && remaining < 30
            && !_heartbeatInFlight && !_heartbeatIdle
            && _lastHeartbeatActivity > Date.now() - 30000) {
          sendHeartbeat();
        }

        // Session countdown — auto sign-out when session expires
        if (remaining <= 0) {
          sessionEl.textContent = 'expired';
          sessionEl.className = 'timer-value critical';
          pillLabel.textContent = 'expired';
          pillLabel.className = 'timer-pill-label critical';
          performSignOut('session-expired');
          return;
        } else {
          sessionEl.textContent = formatCountdown(remaining);
          // Show ▶ on the minimized pill when heartbeat is active (not idle)
          var pillReady = HTML_CONFIG.ENABLE_HEARTBEAT && _heartbeatInterval && !_heartbeatIdle && !_heartbeatInFlight;
          pillLabel.textContent = formatCountdown(remaining) + (pillReady ? ' \u25B6' : '');
          if (remaining <= 30) {
            sessionEl.className = 'timer-value critical';
            pillLabel.className = 'timer-pill-label critical';
            if (!_expiryWarningShown) {
              _expiryWarningShown = true;
              showSessionWarning('Session expiring in ' + formatCountdown(remaining) + ' \u2014 interact with the page to stay signed in.');
            } else if (document.getElementById('session-warning-banner').style.display === 'flex') {
              document.getElementById('session-warning-text').textContent = 'Session expiring in ' + formatCountdown(remaining) + ' \u2014 interact with the page to stay signed in.';
            }
          } else if (remaining <= 60) {
            sessionEl.className = 'timer-value critical';
            pillLabel.className = 'timer-pill-label critical';
          } else if (remaining <= 90) {
            sessionEl.className = 'timer-value warn';
            pillLabel.className = 'timer-pill-label warn';
          } else {
            sessionEl.className = 'timer-value';
            pillLabel.className = 'timer-pill-label';
          }
        }

        // Absolute session countdown — hard ceiling, never resets
        // Phase 6 (H-4): uses _curAbsDur from enforcer (tamper-proof)
        if (_curAbsDur > 0) {
          var absStartTime = parseInt(storage.getItem(ABSOLUTE_START_KEY) || '0', 10);
          var absElapsed = (Date.now() - absStartTime) / 1000;
          var absRemaining = _curAbsDur - absElapsed;
          if (absRemaining <= 0) {
            absoluteEl.textContent = 'expired';
            absoluteEl.className = 'timer-value critical';
            performSignOut('absolute-expired');
            return;
          } else {
            absoluteEl.textContent = formatCountdown(absRemaining);
            if (absRemaining <= 60) {
              absoluteEl.className = 'timer-value critical';
              if (!_absoluteWarningShown) {
                _absoluteWarningShown = true;
                showAbsoluteWarning('Session ending in ' + formatCountdown(absRemaining) + ' \u2014 you will need to sign in again.');
              } else if (document.getElementById('absolute-warning-banner').style.display === 'flex') {
                document.getElementById('absolute-warning-text').textContent = 'Session ending in ' + formatCountdown(absRemaining) + ' \u2014 you will need to sign in again.';
              }
            } else if (absRemaining <= 120) {
              absoluteEl.className = 'timer-value warn';
            } else {
              absoluteEl.className = 'timer-value';
            }
          }
        }

        // Heartbeat countdown — shows time until next heartbeat tick
        if (HTML_CONFIG.ENABLE_HEARTBEAT && _heartbeatInterval && remaining > 0) {
          var sinceLastTick = (Date.now() - _lastHeartbeatTick) / 1000;
          var nextTickIn = (HTML_CONFIG.HEARTBEAT_INTERVAL / 1000) - sinceLastTick;
          if (nextTickIn < 0) nextTickIn = 0;
          var isActive = _lastHeartbeatActivity > (_lastHeartbeatTick - HTML_CONFIG.HEARTBEAT_INTERVAL);
          if (_heartbeatInFlight) {
            heartbeatEl.textContent = 'sending...';
            heartbeatEl.className = 'timer-value warn';
          } else if (_heartbeatIdle) {
            heartbeatEl.textContent = formatCountdown(nextTickIn) + ' (idle)';
            heartbeatEl.className = 'timer-value warn';
          } else {
            // isActive already tells us if the next tick will fire (activity within
            // the current interval window = what the heartbeat interval checks).
            var indicator = isActive ? ' \u25B6 ready' : ' (idle)';
            heartbeatEl.textContent = formatCountdown(nextTickIn) + indicator;
            heartbeatEl.className = isActive ? 'timer-value' : 'timer-value warn';
          }
        } else if (remaining <= 0 && heartbeatEl) {
          heartbeatEl.textContent = 'expired';
          heartbeatEl.className = 'timer-value critical';
        }

      }, 1000);
    }

    function stopCountdownTimers() {
      if (_timerInterval) { clearInterval(_timerInterval); _timerInterval = null; }
      var timersEl = document.getElementById('auth-timers');
      if (timersEl) timersEl.style.display = 'none';
      var storage = getStorage();
      storage.removeItem(SESSION_START_KEY);
      storage.removeItem(ABSOLUTE_START_KEY);
    }

    // =============================================
    // AUTH — Heartbeat (Activity-Based Session Extension)
    // Monitors DOM activity events. On each interval tick, if the
    // user has been active since the last heartbeat, sends a lightweight
    // ?heartbeat=TOKEN request to the GAS server which resets createdAt,
    // extending the session. If no activity → no heartbeat → session
    // expires naturally at SERVER_SESSION_DURATION.
    // =============================================

    var _ACTIVITY_EVENTS = ['mousemove', 'mousedown', 'keydown', 'touchstart', 'scroll'];
    var _heartbeatInterval = null;
    var _lastHeartbeatActivity = 0;  // timestamp of last activity seen by heartbeat
    var _lastHeartbeatTick = 0;      // timestamp of the last interval tick (or start)
    var _heartbeatInFlight = false;  // prevents overlapping heartbeat requests
    var _heartbeatInFlightTimer = null;  // timeout to auto-clear _heartbeatInFlight if response never arrives
    var _heartbeatIdle = false;      // true when last tick found no activity
    var _expiryWarningShown = false;  // prevents repeated session "expiring soon" banner displays
    var _absoluteWarningShown = false; // prevents repeated absolute "expiring soon" banner displays

    function startHeartbeat() {
      if (!HTML_CONFIG.ENABLE_HEARTBEAT) return;

      // Track activity for heartbeat
      _lastHeartbeatActivity = Date.now();
      _lastHeartbeatTick = Date.now();
      _heartbeatIdle = false;
      _ACTIVITY_EVENTS.forEach(function(evt) {
        document.addEventListener(evt, _onHeartbeatActivity, { passive: true });
      });
      _heartbeatInterval = setInterval(function() {
        _lastHeartbeatTick = Date.now();
        // Only send heartbeat if user was active since last tick
        var lastTick = _lastHeartbeatTick - HTML_CONFIG.HEARTBEAT_INTERVAL;
        if (_lastHeartbeatActivity < lastTick) {
          // No activity since last heartbeat tick — skip (let session expire if idle)
          _heartbeatIdle = true;
          return;
        }
        _heartbeatIdle = false;
        sendHeartbeat();
      }, HTML_CONFIG.HEARTBEAT_INTERVAL);
    }

    function _onHeartbeatActivity() {
      _lastHeartbeatActivity = Date.now();
      _heartbeatIdle = false;  // clear idle so display immediately shows ▶ ready
    }

    // Phase 7 (H-5): postMessage-based heartbeat — token never appears in URL
    function sendHeartbeat() {
      if (_heartbeatInFlight) return;  // don't stack requests
      var session = loadSession();
      if (!session.token) return;

      var gasApp = document.getElementById('gas-app');
      var baseUrl = (gasApp && gasApp.dataset.baseUrl) || '';
      if (!gasApp || !baseUrl) return;

      _heartbeatInFlight = true;
      // Safety timeout: if no response arrives within 15s, clear the in-flight
      // flag so future heartbeats aren't permanently blocked by a dropped response
      if (_heartbeatInFlightTimer) clearTimeout(_heartbeatInFlightTimer);
      _heartbeatInFlightTimer = setTimeout(function() {
        _heartbeatInFlight = false;
        _heartbeatInFlightTimer = null;
      }, 15000);
      var hbEl = document.getElementById('timer-heartbeat');
      if (hbEl) {
        hbEl.textContent = 'sending...';
        hbEl.className = 'timer-value warn';
      }

      // Use a hidden iframe so the heartbeat doesn't destroy the main GAS app UI
      var hbFrame = document.getElementById('gas-heartbeat');
      if (!hbFrame) {
        hbFrame = document.createElement('iframe');
        hbFrame.id = 'gas-heartbeat';
        hbFrame.style.cssText = 'display:none;width:0;height:0;border:0';
        document.body.appendChild(hbFrame);
      }
      // Phase 7: Load with action parameter only — NO token in URL
      hbFrame.src = baseUrl + '?action=heartbeat';
      // Token is sent via postMessage after the iframe signals ready
      // (handled by gas-heartbeat-ready message handler below)
    }

    function stopHeartbeat() {
      if (_heartbeatInterval) { clearInterval(_heartbeatInterval); _heartbeatInterval = null; }
      if (_heartbeatInFlightTimer) { clearTimeout(_heartbeatInFlightTimer); _heartbeatInFlightTimer = null; }
      _ACTIVITY_EVENTS.forEach(function(evt) {
        document.removeEventListener(evt, _onHeartbeatActivity);
      });
      _heartbeatInFlight = false;
      var hbFrame = document.getElementById('gas-heartbeat');
      if (hbFrame) hbFrame.remove();
    }

    function performSignOut(reason) {
      var session = loadSession();
      clearSession();
      stopCountdownTimers();
      stopHeartbeat();
      // Hide tab-takeover overlay if visible (sign-out takes priority)
      document.getElementById('tab-takeover-wall').style.display = 'none';
      _tabSurrendered = false;

      // Notify other tabs (sessionStorage/hipaa mode — BroadcastChannel)
      // For localStorage/standard mode, cross-tab sync happens automatically via the storage event.
      if (_signOutChannel) {
        _signOutChannel.postMessage({ type: 'sign-out' });
      }

      // Phase 7 (H-6): Notify GAS server to invalidate server-side session
      // Token sent via postMessage, NOT URL parameters
      if (session.token) {
        var gasApp = document.getElementById('gas-app');
        var baseUrl = (gasApp && gasApp.dataset.baseUrl) || '';
        if (gasApp && baseUrl) {
          var _soToken = session.token;  // capture for closure
          gasApp.src = baseUrl + '?action=signout';
          // Token sent after iframe signals ready (gas-signout-ready handler)
          var _soHandled = false;
          var _soHandler = function(event) {
            if (_soHandled) return;
            if (!event.data || event.data.type !== 'gas-signout-ready') return;
            _soHandled = true;
            window.removeEventListener('message', _soHandler);
            event.source.postMessage({
              type: 'signout-token',
              token: _soToken
            }, event.origin);
          };
          window.addEventListener('message', _soHandler);
          // Cleanup if GAS doesn't respond within 10 seconds
          setTimeout(function() {
            if (!_soHandled) {
              _soHandled = true;
              window.removeEventListener('message', _soHandler);
            }
          }, 10000);
        }
      }

      // Google OAuth access token is not stored client-side (only used during initial exchange)
      // and expires naturally (1 hour). Server-side session invalidation is sufficient.

      var message = 'You have been signed out.';
      if (reason === 'session-expired') {
        message = 'Your session has expired. Click "Sign In with Google" to continue.';
        sessionStorage.setItem(SESSION_EXPIRED_FLAG, '1');
      } else if (reason === 'absolute-expired') {
        var hours = Math.round((_sessionTimerEnforcer.getAbsoluteDuration() || ABSOLUTE_SESSION_DURATION) / 3600);
        message = 'Your ' + hours + '-hour session has ended. Please sign in again.';
        sessionStorage.setItem(SESSION_EXPIRED_FLAG, '1');
      }
      showAuthWall(message);
    }

    function attemptReauth() {
      signInClient = google.accounts.oauth2.initTokenClient({
        client_id: CLIENT_ID,
        scope: 'openid email profile',
        prompt: '',
        callback: function(response) {
          if (response.error) {
            showAbsoluteWarning('Session expired. Click "Sign In" to continue.');
            signInClient = google.accounts.oauth2.initTokenClient({
              client_id: CLIENT_ID,
              scope: 'openid email profile',
              prompt: 'select_account',
              callback: handleTokenResponse
            });
          } else {
            _authNonce = null; // Consume nonce on successful silent re-auth
            exchangeToken(response.access_token);
          }
        }
      });
      _authNonce = Math.random().toString(36).substring(2) + Date.now().toString(36);
      signInClient.requestAccessToken();
    }

    function _renderAdminSessions(sessions) {
      var list = document.getElementById('asp-session-list');
      if (!sessions || sessions.length === 0) {
        list.innerHTML = '<div class="asp-status">No active sessions found.</div>';
        return;
      }
      var html = '';
      for (var i = 0; i < sessions.length; i++) {
        var s = sessions[i];
        var selfClass = s.isSelf ? ' asp-self' : '';
        var roleClass = s.isEmergencyAccess ? ' asp-emergency' : '';
        var selfLabel = s.isSelf ? ' (you)' : '';
        var created = new Date(s.createdAt);
        var lastAct = new Date(s.lastActivity);
        var absMin = Math.floor(s.absoluteRemaining / 60);
        var absSec = s.absoluteRemaining % 60;
        var rollMin = Math.floor(s.rollingRemaining / 60);
        var rollSec = s.rollingRemaining % 60;

        html += '<div class="asp-session' + selfClass + '">'
          + '<div><span class="asp-email">' + _escHtml(s.email) + selfLabel + '</span>'
          + '<span class="asp-role' + roleClass + '">' + _escHtml(s.role)
          + (s.isEmergencyAccess ? ' (emergency)' : '') + '</span></div>'
          + '<div class="asp-meta">Signed in: ' + created.toLocaleTimeString() + '</div>'
          + '<div class="asp-meta">Last activity: ' + lastAct.toLocaleTimeString() + '</div>'
          + '<div class="asp-meta">Remaining: ' + absMin + 'm ' + absSec + 's abs / '
          + rollMin + 'm ' + rollSec + 's rolling</div>';
        if (!s.isSelf) {
          html += '<button class="asp-kick" data-email="' + _escHtml(s.email)
            + '">Sign Out User</button>';
        }
        html += '</div>';
      }
      list.innerHTML = html;

      // Attach click handlers to sign-out buttons
      var kickBtns = list.querySelectorAll('.asp-kick');
      for (var k = 0; k < kickBtns.length; k++) {
        kickBtns[k].addEventListener('click', function() {
          var email = this.getAttribute('data-email');
          this.textContent = 'Signing out...';
          this.disabled = true;
          _adminSignOutUser(email);
        });
      }
    }

    // On page load: check for existing session or start sign-in
    (function() {
      var session = loadSession();
      if (session.token) {
        // Resume existing session — load GAS iframe with session token
        var gasApp = document.getElementById('gas-app');
        var baseUrl = (gasApp && gasApp.dataset.baseUrl) || '';
        if (gasApp && baseUrl) {
          // Do NOT set _expectingSession here — on page load, there is no prior
          // iframe content that could send a stale gas-needs-auth. The flag would
          // cause the REAL gas-needs-auth (from an admin-invalidated session) to be
          // silently dropped, leaving the user stuck on "Reconnecting...".
          // _expectingSession is only needed after mid-session iframe navigations
          // (gas-session-created, Use Here) where prior content may still be loaded.
          gasApp.src = baseUrl + '?session=' + encodeURIComponent(session.token);
        }
        // Defer showApp until gas-auth-ok confirms the session is still valid
        // server-side. Without this, an attacker who injects a forged token into
        // storage can trick the page into hiding the auth wall on reload.
        _pendingSessionShow = { email: session.email, displayName: null, role: session.role };
        showReconnecting();
      } else {
        // No session — the iframe IIFE created the element without navigating it
        // (no src set on auth pages). The auth flow (exchangeViaUrl) will set
        // the iframe src once Google Sign-In obtains a token.

        // If the session expired before this reload (e.g. auto-refresh after
        // timeout), show expired message. Otherwise show generic auth wall.

        sessionStorage.removeItem(SESSION_EXPIRED_FLAG);
        if (sessionWasExpired) {
          showAuthWall('Session expired. Click "Sign In with Google" to continue.');
        } else {
          showAuthWall();
        }
